Q3.1:i) Create a table person with following structure:-
Ans: create table person (
     Id int primary key,
     name varchar(50),
     address varchar(100)
     ); 


ii)Create a table account with following structure:-
Ans: create table account (
     Acid int primary key,
     Bs int,
     Hra int,
     Da int,
     Id int,
     foreign key(Id) references person(Id)
     );

Q3.3:
i. select name from person table, bs, hra and da from account table by performing inner join operation.
Ans: select p.name, a.Bs, a.Hra, a.Da
     from person p
     inner join account a on p.Id = a.Id;

ii. select name from person table, bs, hra and da from account table by performing left join operation.
Ans: select p.name, a.Bs, a.Hra, a.Da
     from person p
     left join account a on p.Id = a.Id;

select name from person table, bs, hra and da from account table by performing right join operation.
Ans: select p.name, a.Bs, a.Hra, a.Da
     from person p
     right join account a on p.Id = a.Id;

iv. select name from person table, bs, hra and da from account table by performing full join operation.
Ans:  select p.name, a.Bs, a.Hra, a.Da
     from person p
     full outer join account a on p.Id = a.Id;
